009    package org.codemonkey.vesijama;
010    
011    import java.util.regex.Pattern;
012    
013    /**
014     * Validates an email address according to <a href="http://www.ietf.org/rfc/rfc2822.txt">RFC 2822</a>, using regular expressions.
015     * <p>
016     * From the original author: <br />
017     * <blockquote>
018     * If you use this code, please keep the author information in tact and reference my site at <a
019     * href="http://www.leshazlewood.com">leshazlewood.com</a>. Thanks!
020     * </blockquote>
021     * <p>
022     * Code sanitized by Benny Bottema (kept validation 100% in tact).
023     * 
024     * @author Les Hazlewood, Benny Bottema
025     */
026    public final class EmailValidationUtil {
027    
028            /**
029             * Private constructor; this is a utility class with static methods only, not designed for extension.
030             */
031            private EmailValidationUtil() {
032                    //
033            }
034    
035            /**
036             * This constant states that domain literals are allowed in the email address, e.g.:
037             * <p>
038             * <tt>someone@[192.168.1.100]</tt> or <br/>
039             * <tt>john.doe@[23:33:A2:22:16:1F]</tt> or <br/>
040             * <tt>me@[my computer]</tt>
041             * </p>
042             * <p>
043             * The RFC says these are valid email addresses, but most people don't like allowing them. If you don't want to
044             * allow them, and only want to allow valid domain names (<a href="http://www.ietf.org/rfc/rfc1035.txt">RFC
045             * 1035</a>, x.y.z.com, etc), change this constant to <tt>false</tt>.
046             * <p>
047             * Its default value is <tt>true</tt> to remain RFC 2822 compliant, but you should set it depending on what you need
048             * for your application.
049             */
050            private static final boolean ALLOW_DOMAIN_LITERALS = false;
051    
052            /**
053             * This contstant states that quoted identifiers are allowed (using quotes and angle brackets around the raw
054             * address) are allowed, e.g.:
055             * <p>
056             * <tt>"John Smith" &lt;john.smith@somewhere.com&gt;</tt>
057             * <p>
058             * The RFC says this is a valid mailbox. If you don't want to allow this, because for example, you only want users
059             * to enter in a raw address (<tt>john.smith@somewhere.com</tt> - no quotes or angle brackets), then change this
060             * constant to <tt>false</tt>.
061             * <p>
062             * Its default value is <tt>true</tt> to remain RFC 2822 compliant, but you should set it depending on what you need
063             * for your application.
064             */
065            private static final boolean ALLOW_QUOTED_IDENTIFIERS = false;
066            // RFC 2822 2.2.2 Structured Header Field Bodies
067            private static final String wsp = "[ \\t]"; // space or tab
068            private static final String fwsp = wsp + "*";
069            // RFC 2822 3.2.1 Primitive tokens
070            private static final String dquote = "\\\"";
071            // ASCII Control characters excluding white space:
072            private static final String noWsCtl = "\\x01-\\x08\\x0B\\x0C\\x0E-\\x1F\\x7F";
073            // all ASCII characters except CR and LF:
074            private static final String asciiText = "[\\x01-\\x09\\x0B\\x0C\\x0E-\\x7F]";
075            // RFC 2822 3.2.2 Quoted characters:
076            // single backslash followed by a text char
077            private static final String quotedPair = "(\\\\" + asciiText + ")";
078            // RFC 2822 3.2.4 Atom:
079            private static final String atext = "[a-zA-Z0-9\\!\\#\\$\\%\\&\\'\\*\\+\\-\\/\\=\\?\\^\\_\\`\\{\\|\\}\\~]";
080            private static final String atom = fwsp + atext + "+" + fwsp;
081            private static final String dotAtomText = atext + "+" + "(" + "\\." + atext + "+)*";
082            private static final String dotAtom = fwsp + "(" + dotAtomText + ")" + fwsp;
083            // RFC 2822 3.2.5 Quoted strings:
084            // noWsCtl and the rest of ASCII except the doublequote and backslash characters:
085            private static final String qtext = "[" + noWsCtl + "\\x21\\x23-\\x5B\\x5D-\\x7E]";
086            private static final String qcontent = "(" + qtext + "|" + quotedPair + ")";
087            private static final String quotedString = dquote + "(" + fwsp + qcontent + ")*" + fwsp + dquote;
088            // RFC 2822 3.2.6 Miscellaneous tokens
089            private static final String word = "((" + atom + ")|(" + quotedString + "))";
090            private static final String phrase = word + "+"; // one or more words.
091            // RFC 1035 tokens for domain names:
092            private static final String letter = "[a-zA-Z]";
093            private static final String letDig = "[a-zA-Z0-9]";
094            private static final String letDigHyp = "[a-zA-Z0-9-]";
095            private static final String rfcLabel = letDig + "(" + letDigHyp + "{0,61}" + letDig + ")?";
096            private static final String rfc1035DomainName = rfcLabel + "(\\." + rfcLabel + ")*\\." + letter + "{2,6}";
097            // RFC 2822 3.4 Address specification
098            // domain text - non white space controls and the rest of ASCII chars not including [, ], or \:
099            private static final String dtext = "[" + noWsCtl + "\\x21-\\x5A\\x5E-\\x7E]";
100            private static final String dcontent = dtext + "|" + quotedPair;
101            private static final String domainLiteral = "\\[" + "(" + fwsp + dcontent + "+)*" + fwsp + "\\]";
102            private static final String rfc2822Domain = "(" + dotAtom + "|" + domainLiteral + ")";
103            private static final String domain = ALLOW_DOMAIN_LITERALS ? rfc2822Domain : rfc1035DomainName;
104            private static final String localPart = "((" + dotAtom + ")|(" + quotedString + "))";
105            private static final String addrSpec = localPart + "@" + domain;
106            private static final String angleAddr = "<" + addrSpec + ">";
107            private static final String nameAddr = "(" + phrase + ")?" + fwsp + angleAddr;
108            private static final String mailbox = nameAddr + "|" + addrSpec;
109            // now compile a pattern for efficient re-use:
110            // if we're allowing quoted identifiers or not:
111            private static final String patternString = ALLOW_QUOTED_IDENTIFIERS ? mailbox : addrSpec;
112            public static final Pattern VALID_PATTERN = Pattern.compile(patternString);
113    
114            public static boolean isValid(final String userEnteredEmailString) {
115                    return VALID_PATTERN.matcher(userEnteredEmailString).matches();
116            }
117    }
